class Sprite
{
  protected $Identificador,
            $Direccion,
            $Posicion_X,
            $Posicion_Y,
 			   		$numero_de_frame = 60;

	//Metodos
	public function update()
        {
            $this->delete();
            $this->move();
            $this->draw();
        }

	private function Borrar_Personaje()
        {
            echo "Jugador borrado <br>";
        }

        private function Mover_Personaje()
        {
            $this->Posicion_X++;
            $this->Posicion_Y++;
            echo "Actualizamos coordenadas: 
                Xpos: " . $this->Posicion_X . " Ypos: " . $this->Posicion_Y . "<br>";
        }

        private function draw()
        {
            echo "Jugador dibujado en nueva posici√≥n <br>";
        }  
}
